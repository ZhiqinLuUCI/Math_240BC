\begin{equation*} \nabla_X e_j=\omega_{ij}(X) e_i\text{,} \end{equation*}

\begin{equation*} R_{ijk\ell}=\langle \nabla_{e_k}\nabla_{e_\ell} e_j-\nabla_{e_\ell}\nabla_{e_k} e_j -\nabla_{[e_k,e_\ell]}e_j,e_i\rangle\text{.} \end{equation*}
