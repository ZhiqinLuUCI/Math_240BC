\begin{equation*} \eta=a_{i_1\cdots i_p}\omega_{i_1}\otimes\cdots\otimes\omega_{i_p}\text{.} \end{equation*}

\begin{equation*} \nabla_{e_k}\eta=\left(e_k(a_{i_1\cdots i_p})-\sum_s a_{i_1\cdots \underset{\underset{sth}{\uparrow}}r\cdots\cdots i_p}\omega_{r\, i_s}(e_k)\right) \omega_{i_1}\otimes\cdots\otimes\omega_{i_p}\text{.} \end{equation*}
