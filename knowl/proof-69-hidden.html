We assume that \(v>0\) on \(B_R\text{.}\) Otherwise we can use \(v+\eps\) instead.

Since \(v\) is a supersolution, \(v^{-1}\) is a subsolution. By the above lemma, we have

\begin{equation*} \underset{B_{\theta R}}{{ ess}\,\sup}\, v^{-p}\leq C\fint_{B_R} v^{-p} dx\text{,} \end{equation*}

which is equivalent to

\begin{equation*} \underset{B_{\theta R}}{{ ess}\,\inf}\, v\geq C \left(\fint_{B_R} v^{-p} dx\right)^{-\frac 1p} \end{equation*}

We write

\begin{align*} \amp \underset{B_{\theta R}}{{ ess}\,\inf}\, v\geq C \left(\fint_{B_R} v^{-p} dx\right)^{-\frac 1p}\\ \amp =C\left(\fint_{B_R}v^{-p} dx\cdot\fint_{B_R}v^p dx\right)^{-1/p}\cdot\left(\fint_{B_R}v^p dx\right)^{1/p}\text{.} \end{align*}

We just need to prove that there exists a \(p_0>0\) such that

\begin{equation*} \fint_{B_R}v^{-p} dx\cdot\fint_{B_R}v^p dx\leq C\text{.} \end{equation*}

We let \(w=\log v-\beta\text{,}\) where \(\beta\) is a constant to be determined. We wish to prove that

\begin{equation} \int_{B_R}e^{p|w|} dx\leq C\tag{9.1.3} \end{equation}

for some \(p>0\text{.}\) If the above inequality is true then we have

\begin{equation*} \int_{B_R}e^{p\log v-p\beta} dx\leq C,\qquad \int_{B_R}e^{-p\log v+p\beta} dx\leq C\text{,} \end{equation*}

and the lemma follows.

The proof of  (9.1.3) is somewhat complicated but the basic idea is still Moser iteration. Let \(v^{-1}\xi^2\) be a test function, where \(\xi \in C_0^\infty(B_R)\text{,}\) we have

\begin{equation*} 0\geq\int a^{ij} v_i(v^{-1}\xi^2)_j=-\int \xi^2a^{ij}w_i w_j+2\int a^{ij} w_i \xi\xi_j\text{.} \end{equation*}

Without loss of generality, we assume that \(R=1\text{.}\) Then we have

\begin{equation*} \int_{B_\sigma}|Dw|^2\leq C \end{equation*}

for any \(0\lt \sigma\lt 1\text{.}\) By choosing suitable \(\beta\text{,}\) we have the Poincaré inequality

\begin{equation*} \int_{B_\sigma}|w|^2 dx\leq C\text{.} \end{equation*}

Let \(\xi\in C_0^\infty(B_1)\) and let \(\phi=\xi^2|w|^{2q}\text{.}\) Then we have

\begin{equation*} \int_{B_\sigma}\xi^2|w|^{2q}|D_i w|^2\leq 2q\int_{B_\sigma}\xi^2|w|^{2q-1} D_i w D_i|w|+2\int_{B_\sigma}\xi|w|^{2q}D_i wD_i \xi\text{.} \end{equation*}

Using the Young's inequality, we have

\begin{equation*} 2q|w|^{2q-1}\leq\frac{2q-1}{2q}|w|^{2q}+(2q)^{2q-1}\text{.} \end{equation*}

Thus we have

\begin{equation*} \frac{1}{2q}\int_{B_\sigma}\xi^2|w|^{2q}|D_iw|^2\leq (2q)^{2q-1}\int_{B_\sigma}|D_i w|^2+4q\int_{B_\sigma}|w|^{2q}|D_i\xi|^2\text{.} \end{equation*}

Finally, we get

\begin{equation*} \int_{B_\sigma}|D(\xi^2|w|^{2q})|^2 dx\leq C(2q)^{2q}+C\tau^{-2}q^2\int_{B_{\sigma+\tau}}|w|^{2q} dx\text{,} \end{equation*}

where \(C\) only depends on \(n\text{,}\) \(\Lambda/\lambda\text{,}\) and \((1-\sigma)^{-1}\text{.}\) Let \(\kappa=n/(n-1)\text{.}\) By Sobolev inequality, we have

\begin{equation*} \left(\int_{B_\sigma}|w|^{2q\kappa} dx\right)^{1/\kappa}\leq C(2q)^{2q}+C\tau^{-2}q^2\int_{B_{\sigma+\tau}}|w|^{2q} dx\text{.} \end{equation*}

We let

\begin{equation*} q_i=\kappa^{i-1},\delta_0=\sigma, \delta_i=\delta_{i-1}-\frac{\sigma-1}{2^i} \end{equation*}

Then

\begin{equation*} \left(\int_{B_{\delta_i}}|w|^{2\kappa^i} dx\right)^{1/\kappa} \leq C(\kappa)^{2(i-1)\kappa^{i-1}}+C(4\kappa)^i\int_{B_{\delta_{i-1}}}|w|^{2\kappa^{i-1}}dx \end{equation*}

Let

\begin{equation*} I_j=\left(\int_{B_{\delta_j}}|w|^{2\kappa^j}\right)^{1/2\kappa^j}\text{.} \end{equation*}

Then

\begin{equation*} I_i\leq C^{1/\kappa^{i-1}}\kappa^{i-1}+C^{1/\kappa^{i-1}}(4\kappa)^{i/\kappa^{i-1}}I_{i-1}\text{.} \end{equation*}

Running the iteration we get

\begin{equation*} I_j\leq C\kappa^j+cI_0\text{.} \end{equation*}

Thus we have

\begin{equation*} ||w||_{L^q(B_1)}\leq Cq\text{.} \end{equation*}

\begin{equation*} \int_{B_1}e^{p|w|}\leq C\text{.} \end{equation*}
