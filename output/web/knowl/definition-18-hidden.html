\begin{equation*} x_{i_1},\cdots, x_{i_p} \end{equation*}

  1. \begin{equation*} \bigcup_{j=1}^{p} B(x_{i_j}, \frac{1}{2\lambda}T_{i_j})\supset \bigcup_{i=1}^{N} B(x_i, \lambda^2R_i); \end{equation*}
